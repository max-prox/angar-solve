Armstrong Number

void solution()
                {
                    int num, temp, digit = 0, k, sum = 0, j;
                    scanf("%d", &num);
                    
                    temp = num;
                    while (temp > 0) {
                        temp /= 10;
                        digit++;
                    }
                
                    temp = num;
                    while (temp > 0) {
                        k = temp % 10;
                        j = 1;
                        for (int i = 0; i < digit; i++) {
                            j *= k;
                        }
                        sum += j;
                        temp /= 10;
                    }
                
                    if (sum == num)
                        printf("Armstrong");
                    else
                        printf("Not Armstrong");
                }
